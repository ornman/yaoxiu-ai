```javascript
// 动画管理器
const AnimationManager = {
  // 入口欢迎动画
  welcome() {
    const tl = gsap.timeline();
    tl.from('.logo-octagon', { scale: 0, rotation: -180, duration: 0.8, ease: 'back.out' })
      .from('.logo-text', { opacity: 0, y: 20, duration: 0.5 }, '-=0.3')
      .from('.xiao-yao-avatar', { scale: 0, duration: 0.5, ease: 'elastic.out' }, '-=0.2')
      .from('.welcome-message', { opacity: 0, y: 10, duration: 0.4 });
  },
  
  // 收藏飞入动画
  favoriteFly(fromEl, toEl) {
    const fromRect = fromEl.getBoundingClientRect();
    const toRect = toEl.getBoundingClientRect();
    
    const flyer = document.createElement('div');
    flyer.className = 'favorite-flyer';
    flyer.innerHTML = '⭐';
    document.body.appendChild(flyer);
    
    gsap.fromTo(flyer, 
      { 
        x: fromRect.left, 
        y: fromRect.top,
        scale: 1 
      },
      { 
        x: toRect.left, 
        y: toRect.top,
        scale: 0.5,
        duration: 0.6,
        ease: 'power2.inOut',
        onComplete: () => {
          flyer.remove();
          this.badgePulse();
        }
      }
    );
  },
  
  // 打字机效果（带音效）
  typewriter(element, text, speed = 30) {
    let index = 0;
    element.textContent = '';
    
    const type = () => {
      if (index < text.length) {
        element.textContent += text.charAt(index);
        index++;
        // 播放轻微音效
        SoundManager.play('type');
        setTimeout(type, speed);
      }
    };
    
    type();
  },
  
  // 粒子庆祝效果
  confetti(x, y, colors = ['#c41e3a', '#d4a574']) {
    for (let i = 0; i < 30; i++) {
      const particle = document.createElement('div');
      particle.className = 'confetti-particle';
      particle.style.cssText = `
        position: fixed;
        left: ${x}px;
        top: ${y}px;
        width: 8px;
        height: 8px;
        background: ${colors[Math.floor(Math.random() * colors.length)]};
        border-radius: 50%;
        pointer-events: none;
      `;
      document.body.appendChild(particle);
      
      gsap.to(particle, {
        x: (Math.random() - 0.5) * 300,
        y: (Math.random() - 0.5) * 300,
        opacity: 0,
        scale: 0,
        duration: 1 + Math.random(),
        ease: 'power2.out',
        onComplete: () => particle.remove()
      });
    }
  }
};
```

```javascript
// ========== 等级系统 ==========
const LevelSystem = {
  getCurrentLevel() {
    return parseInt(localStorage.getItem('yao-level') || '1');
  },
  
  addXP(amount) {
    const currentXP = parseInt(localStorage.getItem('yao-xp') || '0');
    const currentLevel = this.getCurrentLevel();
    const nextLevelXP = this.getXPForLevel(currentLevel + 1);
    
    const newXP = currentXP + amount;
    
    if (newXP >= nextLevelXP) {
      this.levelUp(currentLevel + 1);
    }
    
    localStorage.setItem('yao-xp', newXP);
  },
  
  levelUp(newLevel) {
    localStorage.setItem('yao-level', newLevel);
    AnimationManager.showLevelUp(newLevel);
    this.grantRewards(newLevel);
  }
};

// ========== 盲盒系统 ==========
const BlindBoxSystem = {
  getDailyDrawCount() {
    const today = new Date().toDateString();
    const lastDrawDate = localStorage.getItem('yao-last-draw-date');
    
    if (lastDrawDate !== today) {
      localStorage.setItem('yao-daily-draws', '0');
      localStorage.setItem('yao-last-draw-date', today);
      return this.getMaxDailyDraws();
    }
    
    const used = parseInt(localStorage.getItem('yao-daily-draws') || '0');
    return this.getMaxDailyDraws() - used;
  },
  
  getMaxDailyDraws() {
    const level = LevelSystem.getCurrentLevel();
    return 3 + Math.floor(level / 2); // 等级越高次数越多
  },
  
  draw() {
    const remaining = this.getDailyDrawCount();
    if (remaining <= 0) return null;
    
    // 记录已使用
    const used = this.getMaxDailyDraws() - remaining;
    localStorage.setItem('yao-daily-draws', (used + 1).toString());
    
    // 随机抽取
    return this.randomPattern();
  },
  
  randomPattern() {
    const rand = Math.random();
    let rarity;
    if (rand < 0.01) rarity = 'legendary';
    else if (rand < 0.1) rarity = 'epic';
    else if (rand < 0.4) rarity = 'rare';
    else rarity = 'common';
    
    const pool = PATTERNS[rarity];
    const unlocked = this.getUnlockedPatterns();
    
    // 优先给未解锁的
    const available = pool.filter(p => !unlocked.includes(p.id));
    if (available.length === 0) {
      // 都有了就给重复的（转换成经验值）
      return { ...pool[Math.floor(Math.random() * pool.length)], duplicate: true };
    }
    
    return available[Math.floor(Math.random() * available.length)];
  }
};

// ========== 签到系统 ==========
const CheckinSystem = {
  isCheckedToday() {
    const today = new Date().toDateString();
    return localStorage.getItem('yao-last-checkin') === today;
  },
  
  getStreakDays() {
    return parseInt(localStorage.getItem('yao-checkin-streak') || '0');
  },
  
  checkin() {
    if (this.isCheckedToday()) return false;
    
    const today = new Date().toDateString();
    const yesterday = new Date(Date.now() - 86400000).toDateString();
    const lastCheckin = localStorage.getItem('yao-last-checkin');
    
    let streak = 1;
    if (lastCheckin === yesterday) {
      streak = this.getStreakDays() + 1;
    }
    
    localStorage.setItem('yao-last-checkin', today);
    localStorage.setItem('yao-checkin-streak', streak.toString());
    
    // 发放奖励
    this.grantReward(streak);
    
    return { success: true, streak };
  },
  
  grantReward(streak) {
    // 基础奖励
    BlindBoxSystem.addDrawCount(1);
    LevelSystem.addXP(10);
    
    // 连续奖励
    const milestone = CHECKIN_REWARDS.streak[streak];
    if (milestone) {
      this.grantMilestoneReward(milestone);
    }
  }
};
```
